## 一、设计背景与目标

### 1.1 评估问题汇总

| 优先级 | 问题 | 影响 |
|--------|------|------|
| P0 | 多个页面为占位页 | 关键管理入口有菜单无能力 |
| P0 | 按钮只提示"开发中" | 无法完成单据流转 |
| P0 | 生产看板存在模拟数据 | 基于错误数据做决策 |
| P1 | 登录交互不够顺手 | 效率低，用户第一印象打折 |
| P1 | 条码管理数据链路未闭合 | 扫码作业难落地 |
| P2 | 经营分析深度不足 | 系统更像录单工具 |

### 1.2 设计目标

1. **可用性底线**: 去除占位页、假按钮、假数据
2. **业务闭环**: 打通销售→生产→质检→出库→应收→利润
3. **老板驾驶舱**: 利润、交期、质量、现金流四维看板

```java
package com.duoduo.jxc.service.impl;

import com.alibaba.excel.EasyExcel;
import com.alibaba.excel.context.AnalysisContext;
import com.alibaba.excel.event.AnalysisEventListener;
import com.duoduo.jxc.dto.ImportResult;
import com.duoduo.jxc.entity.ProductSpu;
import com.duoduo.jxc.mapper.ProductSpuMapper;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.web.multipart.MultipartFile;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

@Service
public class ImportExportServiceImpl implements ImportExportService {

    private final ProductSpuMapper productSpuMapper;

    @Override
    @Transactional(rollbackFor = Exception.class)
    public ImportResult importProducts(MultipartFile file, Boolean updateExisting) {
        ImportResult result = new ImportResult();
        result.setErrors(new ArrayList<>());
        List<ProductSpu> successList = new ArrayList<>();
        int[] counter = {0, 0, 0}; // total, success, update

        try {
            EasyExcel.read(file.getInputStream(), ProductImportDTO.class, 
                new AnalysisEventListener<ProductImportDTO>() {
                    @Override
                    public void invoke(ProductImportDTO data, AnalysisContext context) {
                        counter[0]++;
                        try {
                            validateProductData(data, counter[0], result.getErrors());
                            if (result.getErrors().stream()
                                    .noneMatch(e -> e.getRow().equals(counter[0]))) {
                                ProductSpu spu = convertToEntity(data);
                                
                                if (updateExisting) {
                                    ProductSpu existing = productSpuMapper
                                        .selectBySpuCode(spu.getSpuCode());
                                    if (existing != null) {
                                        spu.setId(existing.getId());
                                        productSpuMapper.updateById(spu);
                                        counter[2]++;
                                    } else {
                                        productSpuMapper.insert(spu);
                                        counter[1]++;
                                    }
                                } else {
                                    productSpuMapper.insert(spu);
                                    counter[1]++;
                                }
                            }
                        } catch (Exception e) {
                            ImportResult.ImportError error = new ImportResult.ImportError();
                            error.setRow(counter[0]);
                            error.setMessage(e.getMessage());
                            result.getErrors().add(error);
                        }
                    }

                    @Override
                    public void doAfterAllAnalysed(AnalysisContext context) {
                        result.setTotalCount(counter[0]);
                        result.setSuccessCount(counter[1]);
                        result.setUpdateCount(counter[2]);
                        result.setFailCount(result.getErrors().size());
                    }
                }).sheet().doRead();
        } catch (IOException e) {
            throw new BusinessException("文件读取失败: " + e.getMessage());
        }

        return result;
    }

    private void validateProductData(ProductImportDTO data, int row, 
                                     List<ImportResult.ImportError> errors) {
        if (data.getSpuCode() == null || data.getSpuCode().trim().isEmpty()) {
            ImportResult.ImportError error = new ImportResult.ImportError();
            error.setRow(row);
            error.setField("spuCode");
            error.setMessage("商品编码不能为空");
            errors.add(error);
        }
        if (data.getSpuName() == null || data.getSpuName().trim().isEmpty()) {
            ImportResult.ImportError error = new ImportResult.ImportError();
            error.setRow(row);
            error.setField("spuName");
            error.setMessage("商品名称不能为空");
            errors.add(error);
        }
    }
}
```

```java
package com.duoduo.jxc.service;

import com.duoduo.jxc.dto.PrintPreviewDTO;

public interface PrintService {
    
    PrintPreviewDTO preview(String bizType, Long bizId);
    
    byte[] generatePdf(String bizType, Long bizId, Long templateId);
    
    byte[] generateLabel(String bizType, Long bizId, int copies);
    
    String generateQRCode(String type, String code);
}

@Service
public class PrintServiceImpl implements PrintService {

    private final PrintTemplateMapper templateMapper;
    private final SalesOrderMapper salesOrderMapper;
    private final PurchaseOrderMapper purchaseOrderMapper;

    @Override
    public PrintPreviewDTO preview(String bizType, Long bizId) {
        PrintTemplate template = getTemplate(bizType);
        Object data = getBusinessData(bizType, bizId);
        
        return PrintPreviewDTO.builder()
            .templateContent(template.getTemplateContent())
            .data(data)
            .previewUrl(generatePreviewUrl(bizType, bizId))
            .build();
    }

    @Override
    public byte[] generatePdf(String bizType, Long bizId, Long templateId) {
        PrintTemplate template = templateMapper.selectById(templateId);
        Object data = getBusinessData(bizType, bizId);
        
        return PdfGenerator.generate(template.getTemplateContent(), data);
    }

    private PrintTemplate getTemplate(String bizType) {
        return templateMapper.selectDefaultByBizType(bizType);
    }

    private Object getBusinessData(String bizType, Long bizId) {
        switch (bizType) {
            case "sales_order":
                return salesOrderMapper.selectDetailById(bizId);
            case "purchase_order":
                return purchaseOrderMapper.selectDetailById(bizId);
            case "bundle":
                return bundleMapper.selectDetailById(bizId);
            default:
                throw new BusinessException("不支持的打印类型: " + bizType);
        }
    }
}
```

```java
@Service
public class ProductionDashboardServiceImpl implements ProductionDashboardService {

    private final ProductionOrderMapper orderMapper;
    private final ProcessRecordMapper recordMapper;
    private final QualityCheckMapper qualityCheckMapper;

    @Override
    public ProductionStatistics getStatistics(Long factoryId) {
        ProductionStatistics stats = new ProductionStatistics();
        
        // 生产中订单数
        stats.setInProgressCount(orderMapper.countByStatus(factoryId, "in_progress"));
        
        // 待开工订单数
        stats.setPendingCount(orderMapper.countByStatus(factoryId, "pending"));
        
        // 今日产量
        LocalDate today = LocalDate.now();
        stats.setTodayOutput(recordMapper.sumQuantityByDate(factoryId, today));
        
        // 本周产量
        LocalDate weekStart = today.with(DayOfWeek.MONDAY);
        LocalDate weekEnd = today.with(DayOfWeek.SUNDAY);
        stats.setWeekOutput(recordMapper.sumQuantityByDateRange(factoryId, weekStart, weekEnd));
        
        // 本月产量
        LocalDate monthStart = today.withDayOfMonth(1);
        LocalDate monthEnd = today.withDayOfMonth(today.lengthOfMonth());
        stats.setMonthOutput(recordMapper.sumQuantityByDateRange(factoryId, monthStart, monthEnd));
        
        // 延期订单数
        stats.setDelayedCount(orderMapper.countDelayed(factoryId, today));
        
        // 质检合格率
        stats.setQualityPassRate(qualityCheckMapper.calculatePassRate(factoryId, today));
        
        return stats;
    }

    @Override
    public List<ProductionOrderCard> getInProgressOrders(Long factoryId) {
        List<ProductionOrder> orders = orderMapper.selectByStatus(factoryId, "in_progress");
        
        return orders.stream().map(order -> {
            ProductionOrderCard card = new ProductionOrderCard();
            card.setOrderId(order.getId());
            card.setOrderNo(order.getOrderNo());
            card.setStyleCode(order.getStyleNo());
            card.setStyleName(order.getStyleName());
            card.setTotalQuantity(order.getQuantity());
            card.setCompletedQuantity(order.getCompletedQuantity());
            card.setProgress(calculateProgress(order));
            card.setCurrentProcess(getCurrentProcessName(order));
            card.setPriority(order.getPriority());
            card.setDeadline(order.getDeadline());
            card.setStatus(order.getStatus());
            
            // 计算延期天数
            if (order.getDeadline() != null && LocalDate.now().isAfter(order.getDeadline())) {
                card.setDelayDays((int) ChronoUnit.DAYS.between(order.getDeadline(), LocalDate.now()));
            }
            
            return card;
        }).collect(Collectors.toList());
    }

    private Integer calculateProgress(ProductionOrder order) {
        if (order.getQuantity() == null || order.getQuantity() == 0) return 0;
        return (int) ((order.getCompletedQuantity() * 100.0) / order.getQuantity());
    }
}
```

```java
@Service
public class BarcodeServiceImpl implements BarcodeService {

    private final BarcodeRuleMapper ruleMapper;
    private final BarcodeRecordMapper recordMapper;
    private final ProductSkuMapper skuMapper;
    private final BundleMapper bundleMapper;

    @Override
    @Transactional(rollbackFor = Exception.class)
    public BarcodeGenerateResult generate(BarcodeGenerateRequest request) {
        // 获取规则
        BarcodeRule rule = request.getRuleId() != null 
            ? ruleMapper.selectById(request.getRuleId())
            : getDefaultRule(request.getType());
        
        if (rule == null) {
            throw new BusinessException("未找到条码生成规则");
        }

        // 生成条码
        String barcode = generateBarcodeByRule(rule, request);
        
        // 保存记录
        BarcodeRecord record = new BarcodeRecord();
        record.setBarcode(barcode);
        record.setType(request.getType());
        record.setRuleId(rule.getRuleId());
        record.setStatus("active");
        
        if ("product".equals(request.getType())) {
            record.setRefId(request.getSkuId());
            record.setRefType("sku");
        } else if ("bundle".equals(request.getType())) {
            record.setRefId(request.getBundleId());
            record.setRefType("bundle");
        }
        
        recordMapper.insert(record);
        
        // 生成图片
        String barcodeImage = generateBarcodeImage(barcode);
        String qrCode = generateQRCodeContent(request);
        String qrCodeImage = generateQRCodeImage(qrCode);
        
        return BarcodeGenerateResult.builder()
            .barcode(barcode)
            .barcodeImage(barcodeImage)
            .qrCode(qrCode)
            .qrCodeImage(qrCodeImage)
            .build();
    }

    private String generateBarcodeByRule(BarcodeRule rule, BarcodeGenerateRequest request) {
        StringBuilder sb = new StringBuilder();
        
        // 前缀
        if (rule.getPrefix() != null) {
            sb.append(rule.getPrefix());
        }
        
        // 日期
        if (rule.getDateFormat() != null) {
            String dateStr = LocalDate.now().format(
                DateTimeFormatter.ofPattern(rule.getDateFormat()));
            sb.append(dateStr);
        }
        
        // 序列号
        Long seq = getNextSequence(rule.getRuleId());
        sb.append(String.format("%0" + rule.getSeqLength() + "d", seq));

    }

        BarcodeRecord record = recordMapper.selectByBarcode(code);
        if (record == null) {
            return null;
        }
        
        BarcodeInfo info = new BarcodeInfo();
        BeanUtils.copyProperties(record, info);
        
        // 填充关联信息
        if ("sku".equals(record.getRefType())) {
            ProductSku sku = skuMapper.selectById(record.getRefId());
            info.setSku(sku);
        } else if ("bundle".equals(record.getRefType())) {
            Bundle bundle = bundleMapper.selectById(record.getRefId());
            info.setBundle(bundle);
        }
        
        return info;
    }
```

```java
@Service
public class ProfitAnalysisServiceImpl implements ProfitAnalysisService {

    private final SalesOrderMapper salesOrderMapper;
    private final ProductionCostMapper productionCostMapper;

    @Override
    public List<StyleProfitDTO> getStyleProfitRanking(ProfitQueryDTO query) {
        // 查询销售数据
        List<StyleSalesSummary> salesList = salesOrderMapper.selectStyleSalesSummary(
            query.getStartDate(), query.getEndDate());
        
        // 查询成本数据
        Map<Long, BigDecimal> costMap = productionCostMapper.selectStyleCostMap(
            query.getStartDate(), query.getEndDate())
            .stream()
            .collect(Collectors.toMap(
                StyleCostSummary::getStyleId,
                StyleCostSummary::getTotalCost
            ));
        
        // 计算利润
        return salesList.stream().map(sales -> {
            StyleProfitDTO dto = new StyleProfitDTO();
            dto.setStyleId(sales.getStyleId());
            dto.setStyleNo(sales.getStyleNo());
            dto.setStyleName(sales.getStyleName());
            dto.setSalesQuantity(sales.getQuantity());
            dto.setSalesAmount(sales.getAmount());
            
            BigDecimal cost = costMap.getOrDefault(sales.getStyleId(), BigDecimal.ZERO);
            dto.setCostAmount(cost);
            dto.setGrossProfit(sales.getAmount().subtract(cost));
            
            if (sales.getAmount().compareTo(BigDecimal.ZERO) > 0) {
                dto.setGrossProfitRate(
                    dto.getGrossProfit().divide(sales.getAmount(), 4, RoundingMode.HALF_UP)
                        .multiply(BigDecimal.valueOf(100)));
            }
            
            return dto;
        }).sorted(Comparator.comparing(StyleProfitDTO::getGrossProfit).reversed())
          .limit(query.getLimit() != null ? query.getLimit() : 20)
          .collect(Collectors.toList());
    }

    @Override
    public StyleProfitDetailDTO getStyleProfitDetail(Long styleId, ProfitQueryDTO query) {
        StyleProfitDetailDTO detail = new StyleProfitDetailDTO();
        
        // 汇总数据
        StyleProfitDTO summary = getStyleSummary(styleId, query);
        detail.setSummary(summary);
        
        // 按颜色分析
        detail.setByColor(getProfitByColor(styleId, query));
        
        // 按尺码分析
        detail.setBySize(getProfitBySize(styleId, query));
        
        // 按月度趋势
        detail.setByMonth(getProfitByMonth(styleId, query));
        
        return detail;
    }
```

```java
@Service
public class CapacityAlertServiceImpl implements CapacityAlertService {

    private final CapacityAlertRuleMapper ruleMapper;
    private final CapacityAlertMapper alertMapper;
    private final ProductionOrderMapper orderMapper;
    private final ProcessCapacityConfigMapper capacityMapper;

    @Scheduled(cron = "0 0 * * * ?")  // 每小时检查一次
    @Override
    public void checkAndGenerateAlerts() {
        List<CapacityAlertRule> rules = ruleMapper.selectActiveRules();
        
        for (CapacityAlertRule rule : rules) {
            checkRule(rule);
        }
    }

    private void checkRule(CapacityAlertRule rule) {
        switch (rule.getMetricType()) {
            case "utilization":
                checkUtilization(rule);
                break;
            case "backlog":
                checkBacklog(rule);
                break;
            case "delay":
                checkDelay(rule);
                break;
        }
    }

    private void checkUtilization(CapacityAlertRule rule) {
        // 计算当前产能利用率
        BigDecimal utilization = calculateUtilization(
            rule.getFactoryId(), rule.getWorkGroupId(), rule.getProcessId());
        
        if (utilization.compareTo(rule.getCriticalThreshold()) >= 0) {
            createAlert(rule, "critical", utilization, rule.getCriticalThreshold());
        } else if (utilization.compareTo(rule.getWarningThreshold()) >= 0) {
            createAlert(rule, "warning", utilization, rule.getWarningThreshold());
        }
    }

    private void checkBacklog(CapacityAlertRule rule) {
        // 计算积压量
        Integer backlog = calculateBacklog(
            rule.getFactoryId(), rule.getWorkGroupId(), rule.getProcessId());
        
        // 获取日产能
        Integer dailyCapacity = getDailyCapacity(rule.getProcessId(), rule.getWorkGroupId());
        
        // 计算积压天数
        BigDecimal backlogDays = dailyCapacity > 0 
            ? BigDecimal.valueOf(backlog).divide(BigDecimal.valueOf(dailyCapacity), 2, RoundingMode.HALF_UP)
            : BigDecimal.valueOf(Integer.MAX_VALUE);
        
        if (backlogDays.compareTo(rule.getCriticalThreshold()) >= 0) {
            createAlert(rule, "critical", backlogDays, rule.getCriticalThreshold());
        } else if (backlogDays.compareTo(rule.getWarningThreshold()) >= 0) {
            createAlert(rule, "warning", backlogDays, rule.getWarningThreshold());
        }
    }

    @Override
    public List<DelayRiskDTO> getDelayRiskOrders(Long factoryId) {
        List<ProductionOrder> orders = orderMapper.selectInProgress(factoryId);
        LocalDate today = LocalDate.now();
        
        return orders.stream()
            .map(order -> calculateDelayRisk(order, today))
            .filter(dto -> !"low".equals(dto.getRiskLevel()))
            .sorted(Comparator.comparing(DelayRiskDTO::getDelayProbability).reversed())
            .collect(Collectors.toList());
    }

    private DelayRiskDTO calculateDelayRisk(ProductionOrder order, LocalDate today) {
        DelayRiskDTO dto = new DelayRiskDTO();
        dto.setOrderId(order.getId());
        dto.setOrderNo(order.getOrderNo());
        dto.setStyleNo(order.getStyleNo());
        dto.setStyleName(order.getStyleName());
        dto.setDeadline(order.getDeadline());
        
        // 计算剩余天数
        int remainingDays = (int) ChronoUnit.DAYS.between(today, order.getDeadline());
        dto.setRemainingDays(remainingDays);
        
        // 计算当前进度
        int progress = order.getQuantity() > 0 
            ? (int) (order.getCompletedQuantity() * 100.0 / order.getQuantity()) 
            : 0;
        dto.setProgress(progress);
        
        // 分析工序瓶颈
        List<ProcessBottleneckDTO> bottlenecks = analyzeBottlenecks(order);
        dto.setBottlenecks(bottlenecks);
        
        // 计算延期概率
        BigDecimal delayProbability = calculateDelayProbability(order, remainingDays, bottlenecks);
        dto.setDelayProbability(delayProbability);
        
        // 判断风险等级
        if (delayProbability.compareTo(BigDecimal.valueOf(0.7)) >= 0) {
            dto.setRiskLevel("high");
            dto.setSuggestion("建议立即加派人手或调整排程");
        } else if (delayProbability.compareTo(BigDecimal.valueOf(0.4)) >= 0) {
            dto.setRiskLevel("medium");
            dto.setSuggestion("建议关注进度，必要时提前干预");
        } else {
            dto.setRiskLevel("low");
            dto.setSuggestion("正常推进即可");
        }
        
        return dto;
    }

    private BigDecimal calculateDelayProbability(ProductionOrder order, int remainingDays, 
                                                  List<ProcessBottleneckDTO> bottlenecks) {
        if (remainingDays < 0) {
            return BigDecimal.ONE;
        }
        
        // 基于剩余工序和产能计算
        int totalEstimatedDays = bottlenecks.stream()
            .mapToInt(ProcessBottleneckDTO::getEstimatedDays)
            .sum();
        
        if (remainingDays >= totalEstimatedDays) {
            return BigDecimal.ZERO;
        }
        
        return BigDecimal.valueOf(totalEstimatedDays - remainingDays)
            .divide(BigDecimal.valueOf(totalEstimatedDays), 2, RoundingMode.HALF_UP);
    }
}
```

```java
@Service
public class QualityCostServiceImpl implements QualityCostService {

    private final QualityCostMapper costMapper;
    private final QualityCostDetailMapper detailMapper;
    private final ReworkOrderMapper reworkMapper;
    private final ProcessRecordMapper recordMapper;

    @Override
    public QualityCostSummaryDTO getSummary(String periodType, String periodValue, Long factoryId) {
        QualityCost cost = costMapper.selectByPeriod(periodType, periodValue, factoryId);
        
        if (cost == null) {
            return new QualityCostSummaryDTO();
        }
        
        QualityCostSummaryDTO dto = new QualityCostSummaryDTO();
        BeanUtils.copyProperties(cost, dto);
        
        // 计算各项占比
        BigDecimal total = cost.getTotalQualityCost();
        if (total.compareTo(BigDecimal.ZERO) > 0) {
            dto.setPreventionRatio(cost.getPreventionCost().divide(total, 4, RoundingMode.HALF_UP).multiply(BigDecimal.valueOf(100)));
            dto.setAppraisalRatio(cost.getAppraisalCost().divide(total, 4, RoundingMode.HALF_UP).multiply(BigDecimal.valueOf(100)));
            dto.setInternalFailureRatio(cost.getInternalFailureCost().divide(total, 4, RoundingMode.HALF_UP).multiply(BigDecimal.valueOf(100)));
            dto.setExternalFailureRatio(cost.getExternalFailureCost().divide(total, 4, RoundingMode.HALF_UP).multiply(BigDecimal.valueOf(100)));
        }
        
        return dto;
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void calculateMonthlyCost(String periodValue, Long factoryId) {
        // 获取当月所有返工单
        List<ReworkOrder> reworks = reworkMapper.selectByPeriod(periodValue, factoryId);
        
        BigDecimal reworkCost = BigDecimal.ZERO;
        BigDecimal scrapCost = BigDecimal.ZERO;
        
        for (ReworkOrder rework : reworks) {
            // 计算返工成本
            BigDecimal reworkAmount = calculateReworkAmount(rework);
            reworkCost = reworkCost.add(reworkAmount);
            
            // 如果有报废
            if (rework.getScrapQuantity() != null && rework.getScrapQuantity() > 0) {
                BigDecimal scrapAmount = calculateScrapAmount(rework);
                scrapCost = scrapCost.add(scrapAmount);
            }
            
            // 记录明细
            saveCostDetail(rework, reworkAmount, scrapAmount);
        }
        
        // 汇总保存
        QualityCost cost = new QualityCost();
        cost.setPeriodType("monthly");
        cost.setPeriodValue(periodValue);
        cost.setFactoryId(factoryId);
        cost.setReworkCost(reworkCost);
        cost.setScrapCost(scrapCost);
        cost.setInternalFailureCost(reworkCost.add(scrapCost));
        cost.setTotalQualityCost(reworkCost.add(scrapCost));
        
        costMapper.insertOrUpdate(cost);
    }

    private BigDecimal calculateReworkAmount(ReworkOrder rework) {
        // 返工成本 = 返工工序工价 × 返工数量
        List<ProcessRecord> records = recordMapper.selectByReworkId(rework.getReworkId());
        return records.stream()
            .map(ProcessRecord::getAmount)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    }
}
```

```java
package com.duoduo.jxc.aspect;

import com.duoduo.jxc.annotation.DataScope;
import com.duoduo.jxc.security.SecurityUtils;
import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Before;
import org.springframework.stereotype.Component;

@Aspect
@Component
public class DataScopeAspect {

    @Before("@annotation(dataScope)")
    public void doBefore(JoinPoint point, DataScope dataScope) {
        Object params = point.getArgs()[0];
        if (params instanceof BaseQueryDTO) {
            BaseQueryDTO query = (BaseQueryDTO) params;
            
            // 根据用户数据权限设置查询条件
            Long userId = SecurityUtils.getCurrentUserId();
            String dataScopeType = SecurityUtils.getDataScopeType();
            
            switch (dataScopeType) {
                case "all":
                    // 全部数据，不加限制
                    break;
                case "department":
                    // 本部门数据
                    query.setDeptId(SecurityUtils.getDeptId());
                    break;
                case "self":
                    // 仅本人数据
                    query.setUserId(userId);
                    break;
            }
        }
    }
}
```

### 7.3 组件库封装

```typescript
// frontend/src/components/table/TableActions.vue
<template>
  <div class="table-actions">
    <el-button 
      v-for="action in visibleActions" 
      :key="action.key"
      :type="action.type || 'primary'"
      :link="action.link"
      :disabled="action.disabled"
      :loading="action.loading"
      @click="handleAction(action)"
    >
      <el-icon v-if="action.icon"><component :is="action.icon" /></el-icon>
      {{ action.label }}
    </el-button>
    
    <el-dropdown v-if="moreActions.length > 0" trigger="click">
      <el-button type="primary" link>
        更多<el-icon class="el-icon--right"><ArrowDown /></el-icon>
      </el-button>
      <template #dropdown>
        <el-dropdown-menu>
          <el-dropdown-item 
            v-for="action in moreActions" 
            :key="action.key"
            @click="handleAction(action)"
          >
            {{ action.label }}
          </el-dropdown-item>
        </el-dropdown-menu>
      </template>
    </el-dropdown>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'

interface TableAction {
  key: string
  label: string
  icon?: string
  type?: 'primary' | 'success' | 'warning' | 'danger' | 'info'
  link?: boolean
  disabled?: boolean
  loading?: boolean
  visible?: boolean
  permission?: string
}

const props = withDefaults(defineProps<{
  actions: TableAction[]
  maxVisible?: number
}>(), {
  maxVisible: 3
})

const visibleActions = computed(() => {
  return props.actions
    .filter(a => a.visible !== false && hasPermission(a.permission))
    .slice(0, props.maxVisible)
})

const moreActions = computed(() => {
  return props.actions
    .filter(a => a.visible !== false && hasPermission(a.permission))
    .slice(props.maxVisible)
})

const emit = defineEmits<{
  action: [action: TableAction]
}>()

const handleAction = (action: TableAction) => {
  emit('action', action)
}

const hasPermission = (permission?: string) => {
  if (!permission) return true
  // 检查权限
  return true
}
</script>
```

### C. 前端代码规范

1. 组件命名：PascalCase，如 `ImportDialog.vue`
2. 接口文件：camelCase，如 `profit.ts`
3. 类型定义：与后端DTO对应，使用interface
4. 样式：使用scoped，避免全局污染
5. 权限控制：使用 `v-permission` 指令或路由守卫
