```sql
  ADD COLUMN image_urls TEXT,
  ADD COLUMN purchase_price DECIMAL(10,2),
  ADD COLUMN retail_price DECIMAL(10,2);
-- 其他7个缺失字段...
```

### 尽快修复 (P1)
```sql
-- 1. Style表
ALTER TABLE jxc_style 
  ADD COLUMN designer_id BIGINT,
  ADD COLUMN target_age_group VARCHAR(50),
  ADD COLUMN tech_pack VARCHAR(500);

-- 2. SizeCategory表
ALTER TABLE jxc_size_category CHANGE COLUMN category_name name VARCHAR(100);

-- 3. 统一菜单权限标识
UPDATE jxc_sys_menu SET perms = 'bom:view' WHERE perms = 'data:bom:view';
-- 其他权限标识...
```
