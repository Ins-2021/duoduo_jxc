**计件工资计算SQL：**

```sql
WITH piece_price AS (
    SELECT 
        pp.style_id,
        pp.process_code,
        pp.employee_id,
        pp.unit_price,
        'employee' AS price_type
    FROM jxc_piece_price pp
    WHERE pp.employee_id = :employeeId
      AND pp.is_deleted = 0
    
    UNION ALL
    
    SELECT 
        pp.style_id,
        pp.process_code,
        NULL AS employee_id,
        pp.unit_price,
        'standard' AS price_type
    FROM jxc_piece_price pp
    WHERE pp.style_id = :styleId
      AND pp.process_code = :processCode
      AND pp.employee_id IS NULL
      AND pp.is_deleted = 0
)
SELECT 
    pr.id,
    pr.employee_id,
    pr.employee_name,
    pr.style_id,
    pr.style_code,
    pr.style_name,
    pr.process_code,
    pr.process_name,
    pr.quantity,
    COALESCE(
        (SELECT unit_price FROM piece_price WHERE price_type = 'employee' LIMIT 1),
        (SELECT unit_price FROM piece_price WHERE price_type = 'standard' LIMIT 1),
        0
    ) AS unit_price,
    pr.quantity * COALESCE(
        (SELECT unit_price FROM piece_price WHERE price_type = 'employee' LIMIT 1),
        (SELECT unit_price FROM piece_price WHERE price_type = 'standard' LIMIT 1),
        0
    ) AS wage_amount
FROM jxc_piece_record pr
WHERE pr.is_deleted = 0
  AND pr.record_date BETWEEN :startDate AND :endDate
  AND pr.audit_status = 'approved';
```

#### 37.2.4 计件工资月度汇总

**月度汇总SQL：**

```sql
SELECT 
    pr.employee_id,
    e.employee_code,
    e.employee_name,
    d.department_name,
    pr.style_id,
    s.style_code,
    s.style_name,
    pr.process_code,
    pr.process_name,
    SUM(pr.quantity) AS total_quantity,
    AVG(pr.unit_price) AS avg_unit_price,
    SUM(pr.wage_amount) AS total_wage,
    COUNT(DISTINCT pr.record_date) AS work_days
FROM jxc_piece_record pr
JOIN jxc_employee e ON pr.employee_id = e.id
LEFT JOIN jxc_department d ON e.department_id = d.id
JOIN jxc_style s ON pr.style_id = s.id
WHERE pr.is_deleted = 0
  AND pr.record_date BETWEEN :startDate AND :endDate
  AND pr.audit_status = 'approved'
GROUP BY pr.employee_id, e.employee_code, e.employee_name, d.department_name,
         pr.style_id, s.style_code, s.style_name, pr.process_code, pr.process_name
ORDER BY e.employee_code, s.style_code, pr.process_code;
```

**个人所得税计算SQL：**

```sql
CREATE FUNCTION calculate_tax(
    p_taxable_income DECIMAL(18,2)
) RETURNS DECIMAL(18,2)
DETERMINISTIC
BEGIN
    DECLARE v_tax DECIMAL(18,2) DEFAULT 0;
    
    IF p_taxable_income <= 36000 THEN
        SET v_tax = p_taxable_income * 0.03;
    ELSEIF p_taxable_income <= 144000 THEN
        SET v_tax = p_taxable_income * 0.10 - 2520;
    ELSEIF p_taxable_income <= 300000 THEN
        SET v_tax = p_taxable_income * 0.20 - 16920;
    ELSEIF p_taxable_income <= 420000 THEN
        SET v_tax = p_taxable_income * 0.25 - 31920;
    ELSEIF p_taxable_income <= 660000 THEN
        SET v_tax = p_taxable_income * 0.30 - 52920;
    ELSEIF p_taxable_income <= 960000 THEN
        SET v_tax = p_taxable_income * 0.35 - 85920;
    ELSE
        SET v_tax = p_taxable_income * 0.45 - 181920;
    END IF;
    
    RETURN GREATEST(0, v_tax);
END;
```

**银行代发文件生成SQL：**

```sql
SELECT 
    ROW_NUMBER() OVER (ORDER BY e.employee_code) AS seq_no,
    e.bank_account,
    e.employee_name,
    FORMAT(ws.actual_amount, 2) AS amount,
    '01' AS id_type,
    e.id_card,
    '工资' AS purpose,
    e.mobile
FROM jxc_wage_sheet ws
JOIN jxc_employee e ON ws.employee_id = e.id
WHERE ws.is_deleted = 0
  AND ws.sheet_status = 'approved'
  AND ws.account_period = :accountPeriod
  AND e.payment_method = 'bank'
  AND e.bank_account IS NOT NULL
ORDER BY e.employee_code;
```

#### 37.5.2 工资凭证自动生成

**凭证生成SQL：**

```sql
INSERT INTO jxc_voucher (
    voucher_no, voucher_date, voucher_type, account_period,
    debit_amount, credit_amount, voucher_status,
    source_type, source_id, create_time, create_by
)
SELECT 
    CONCAT('ZZ', DATE_FORMAT(NOW(), '%Y%m%d'), LPAD(IFNULL(MAX(v.id), 0) + 1, 4, '0')),
    NOW(),
    'transfer',
    ws.account_period,
    ws.total_amount,
    ws.total_amount,
    'draft',
    'wage',
    ws.id,
    NOW(),
    :userId
FROM jxc_voucher v
RIGHT JOIN jxc_wage_sheet ws ON 1=1
WHERE ws.id = :wageSheetId
GROUP BY ws.id;

INSERT INTO jxc_voucher_detail (
    voucher_id, account_code, account_name, 
    debit_amount, credit_amount, summary
)
SELECT 
    v.id,
    acc.account_code,
    acc.account_name,
    CASE WHEN d.direction = 'debit' THEN d.amount ELSE 0 END,
    CASE WHEN d.direction = 'credit' THEN d.amount ELSE 0 END,
    d.summary
FROM jxc_voucher v
JOIN (
    SELECT 'debit' AS direction, '5001.01' AS account_code, 
           SUM(wage_amount) AS amount, '计提直接人工' AS summary
    FROM jxc_wage_sheet_detail
    WHERE wage_sheet_id = :wageSheetId AND cost_type = 'production'
    
    UNION ALL
    
    SELECT 'debit', '5101.01', SUM(wage_amount), '计提车间管理人员工资'
    FROM jxc_wage_sheet_detail
    WHERE wage_sheet_id = :wageSheetId AND cost_type = 'workshop'
    
    UNION ALL
    
    SELECT 'credit', '2211.01', SUM(total_amount), '应付职工薪酬'
    FROM jxc_wage_sheet
    WHERE id = :wageSheetId
) d ON 1=1
JOIN jxc_account acc ON d.account_code = acc.account_code
WHERE v.source_id = :wageSheetId;
```
