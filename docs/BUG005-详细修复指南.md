# B005 Bug详细修复指南

## 问题描述
多个Service实现不完整，仅继承ServiceImpl，缺少CRUD方法、参数校验、事务控制

## 涉及文件（10个基础Service）
1. `StyleServiceImpl.java`
2. `StyleColorwayServiceImpl.java`
3. `ColorServiceImpl.java`
4. `SizeCategoryServiceImpl.java`
5. `SizeOptionServiceImpl.java`
6. `SizeChartServiceImpl.java`
7. `SizeRatioTemplateServiceImpl.java`
8. `FactoryServiceImpl.java`
9. `WorkshopServiceImpl.java`
10. `WorkGroupServiceImpl.java`

## 修复标准

### 每个Service必须包含的方法
1. **pageQuery** - 分页查询
2. **getDetail** - 获取详情
3. **create** - 创建
4. **update** - 更新
5. **delete** - 删除

### 修复示例（以StyleServiceImpl为例）

```java
package com.duoduo.jxc.service.impl;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.duoduo.jxc.common.BizCode;
import com.duoduo.jxc.common.PageResult;
import com.duoduo.jxc.entity.Style;
import com.duoduo.jxc.exception.BusinessException;
import com.duoduo.jxc.mapper.StyleMapper;
import com.duoduo.jxc.query.StyleQuery;
import com.duoduo.jxc.service.StyleService;
import com.duoduo.jxc.vo.StyleVO;
import lombok.RequiredArgsConstructor;
import org.springframework.beans.BeanUtils;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@RequiredArgsConstructor
public class StyleServiceImpl extends ServiceImpl<StyleMapper, Style> implements StyleService {

    /**
     * 分页查询
     */
    @Override
    public PageResult<StyleVO> pageQuery(StyleQuery query) {
        Page<Style> page = lambdaQuery()
                .eq(query.getStatus() != null, Style::getStatus, query.getStatus())
                .like(query.getKeyword() != null, Style::getStyleNo, query.getKeyword())
                .orderByDesc(Style::getCreateTime)
                .page(new Page<>(query.getPageNum(), query.getPageSize()));
        
        // 转换为VO
        return PageResult.of(page.getRecords(), page.getTotal(), StyleVO.class);
    }

    /**
     * 获取详情
     */
    @Override
    public StyleVO getDetail(Long id) {
        Style style = getById(id);
        if (style == null) {
            throw new BusinessException(BizCode.DATA_NOT_FOUND);
        }
        StyleVO vo = new StyleVO();
        BeanUtils.copyProperties(style, vo);
        return vo;
    }

    /**
     * 创建
     */
    @Override
    @Transactional(rollbackFor = Exception.class)
    public Long create(StyleDTO dto) {
        // 参数校验
        if (dto.getStyleNo() == null || dto.getStyleNo().trim().isEmpty()) {
            throw new BusinessException(BizCode.PARAM_ERROR, "款式编号不能为空");
        }
        
        // 检查重复
        Long count = lambdaQuery()
                .eq(Style::getStyleNo, dto.getStyleNo())
                .count();
        if (count > 0) {
            throw new BusinessException(BizCode.BUSINESS_ERROR, "款式编号已存在");
        }
        
        Style style = new Style();
        BeanUtils.copyProperties(dto, style);
        style.setStatus(0); // 默认状态
        save(style);
        return style.getId();
    }

    /**
     * 更新
     */
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void update(StyleDTO dto) {
        if (dto.getId() == null) {
            throw new BusinessException(BizCode.PARAM_ERROR, "ID不能为空");
        }
        
        Style exist = getById(dto.getId());
        if (exist == null) {
            throw new BusinessException(BizCode.DATA_NOT_FOUND);
        }
        
        // 如果修改了编号，检查重复
        if (!exist.getStyleNo().equals(dto.getStyleNo())) {
            Long count = lambdaQuery()
                    .eq(Style::getStyleNo, dto.getStyleNo())
                    .ne(Style::getId, dto.getId())
                    .count();
            if (count > 0) {
                throw new BusinessException(BizCode.BUSINESS_ERROR, "款式编号已存在");
            }
        }
        
        BeanUtils.copyProperties(dto, exist);
        updateById(exist);
    }

    /**
     * 删除
     */
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void delete(Long id) {
        Style exist = getById(id);
        if (exist == null) {
            throw new BusinessException(BizCode.DATA_NOT_FOUND);
        }
        
        // 可选：检查是否被引用，如果被引用则不允许删除
        
        removeById(id);
    }
}
```

## Service接口定义
```java
package com.duoduo.jxc.service;

import com.duoduo.jxc.common.PageResult;
import com.duoduo.jxc.entity.Style;
import com.duoduo.jxc.query.StyleQuery;
import com.duoduo.jxc.vo.StyleVO;

public interface StyleService extends IService<Style> {
    
    PageResult<StyleVO> pageQuery(StyleQuery query);
    
    StyleVO getDetail(Long id);
    
    Long create(StyleDTO dto);
    
    void update(StyleDTO dto);
    
    void delete(Long id);
}
```

## 简化批量修复方案

如果10个Service结构类似，可以使用以下批量修复模式：

```java
// 通用模板
@Service
@RequiredArgsConstructor
public class XxxServiceImpl extends ServiceImpl<XxxMapper, Xxx> implements XxxService {

    @Override
    public PageResult<XxxVO> pageQuery(XxxQuery query) {
        Page<Xxx> page = lambdaQuery()
                .eq(query.getStatus() != null, Xxx::getStatus, query.getStatus())
                .like(query.getKeyword() != null, Xxx::getName, query.getKeyword())
                .orderByDesc(Xxx::getCreateTime)
                .page(new Page<>(query.getPageNum(), query.getPageSize()));
        return PageResult.of(page.getRecords(), page.getTotal(), XxxVO.class);
    }

    @Override
    public XxxVO getDetail(Long id) {
        Xxx entity = getById(id);
        if (entity == null) {
            throw new BusinessException(BizCode.DATA_NOT_FOUND);
        }
        XxxVO vo = new XxxVO();
        BeanUtils.copyProperties(entity, vo);
        return vo;
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public Long create(XxxDTO dto) {
        // 参数校验：名称不能为空
        if (dto.getName() == null || dto.getName().trim().isEmpty()) {
            throw new BusinessException(BizCode.PARAM_ERROR, "名称不能为空");
        }
        
        // 检查名称重复
        Long count = lambdaQuery().eq(Xxx::getName, dto.getName()).count();
        if (count > 0) {
            throw new BusinessException(BizCode.BUSINESS_ERROR, "名称已存在");
        }
        
        Xxx entity = new Xxx();
        BeanUtils.copyProperties(dto, entity);
        entity.setStatus(0);
        save(entity);
        return entity.getId();
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void update(XxxDTO dto) {
        if (dto.getId() == null) {
            throw new BusinessException(BizCode.PARAM_ERROR, "ID不能为空");
        }
        
        Xxx exist = getById(dto.getId());
        if (exist == null) {
            throw new BusinessException(BizCode.DATA_NOT_FOUND);
        }
        
        // 检查名称重复
        if (!exist.getName().equals(dto.getName())) {
            Long count = lambdaQuery()
                    .eq(Xxx::getName, dto.getName())
                    .ne(Xxx::getId, dto.getId())
                    .count();
            if (count > 0) {
                throw new BusinessException(BizCode.BUSINESS_ERROR, "名称已存在");
            }
        }
        
        BeanUtils.copyProperties(dto, exist);
        updateById(exist);
    }

    @Override
    @Transactional(rollbackFor = Exception.class)
    public void delete(Long id) {
        Xxx exist = getById(id);
        if (exist == null) {
            throw new BusinessException(BizCode.DATA_NOT_FOUND);
        }
        removeById(id);
    }
}
```
